Usage - Copy this snippet

Add this script to your application to automatically capture console output:

<script src="abbacchio-console.js"></script>
<script>
  AbbacchioConsole.init({
    url: 'http://localhost:4000/api/logs',
    channel: 'my-app',
    appName: 'my-web-app',
    // secretKey: 'optional-encryption-key',
  });
</script>
